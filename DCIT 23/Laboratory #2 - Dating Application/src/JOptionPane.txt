// Swing
// Frame
JFrame frame = new JFrame();
frame.setAlwaysOnTop(true);

// Fields
String[] values = {"Option 1","Option 2","Option 3"};

// Message Dialog
JOptionPane.showMessageDialog(frame, "Message", "Title", JOptionPane.INFORMATION_MESSAGE);

// Input Message
JOptionPane.showInputDialog(frame, "Message", "Title", JOptionPane.INFORMATION_MESSAGE);

// Input Selection
JOptionPane.showInputDialog(frame, "Message", "Title", JOptionPane.INFORMATION_MESSAGE, null, values, values[0]);
